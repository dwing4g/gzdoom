class SectorSilencer : Actor
{
	default
	{
		+NOBLOCKMAP
		+NOGRAVITY
		+DONTSPLASH
		RenderStyle "None";
	}
	
	override void BeginPlay ()
	{
		Super.BeginPlay ();
		CurSector.Flags |= Sector.SECF_SILENT;
	}

	override void OnDestroy ()
	{
		if (CurSector != null)
		{
			CurSector.Flags &= ~Sector.SECF_SILENT;
		}
		Super.OnDestroy();
	}
}
